import java.awt.Dimension;
import java.awt.MouseInfo;
import java.awt.Point;
import java.awt.Robot;
import java.awt.Toolkit;
import java.util.Random;

public class Main {
    private static long t1 = System.currentTimeMillis();
    private static Point p1 = MouseInfo.getPointerInfo().getLocation();
    private static final int V1 = 30000;

    public static void main(String[] args) {
        try {
            var r = new Robot();
            var rd = new Random();
            var d = Toolkit.getDefaultToolkit().getScreenSize();

            while (true) {
                Point cp = MouseInfo.getPointerInfo().getLocation();
                long n = System.currentTimeMillis();

                if (!cp.equals(p1)) {
                    t1 = n;
                    p1 = cp;
                }

                if (n - t1 >= V1) {
                    int x = rd.nextInt(d.width);
                    int y = rd.nextInt(d.height);
                    r.mouseMove(x, y);
                    p1 = new Point(x, y);
                    t1 = System.currentTimeMillis();
                }

                Thread.sleep(1000);
            }
        } catch (Exception e) {
            // silent
        }
    }
}
