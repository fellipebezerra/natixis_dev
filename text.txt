          v-bind="inputAttrs"
          v-on="inputHandlers"

        />
        <!-- Dropdown de sugestões -->

    </template>

onCurrencyInput() {
  // apenas para reagir ao input — o computed já filtra
},

pickCurrency(option, addTag) {
  // Guarda o text sem "/" para a tag, mas o value (ID) para o submit
  addTag(option.text.replace('/', ''));  // ex: AUDUSD em vez de AUD/USD
  this.currencySearch = '';
},

selectCurrency(addTag) {
  if (this.filteredCurrencies.length > 0) {
    this.pickCurrency(this.filteredCurrencies[0], addTag);
  }
},

pickTradingGroup(option, addTag) {
  addTag(option.text);
  this.tradingGroupSearch = '';
},

selectTradingGroup(addTag) {
  if (this.filteredTradingGroups.length > 0) {
    this.pickTradingGroup(this.filteredTradingGroups[0], addTag);
  }
},
